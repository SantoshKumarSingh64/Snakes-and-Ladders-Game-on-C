void start()
{
  int X=196,i;
  for(i=0;i<5;i++)
  {
   paint1();
   setfillstyle(SOLID_FILL,BLUE);
   setcolor(YELLOW);
   rectangle(120,300,X,320);
   floodfill(125,310,YELLOW);
   delay(1000);
   X=X+76;
  }
}
